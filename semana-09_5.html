Paso 2: Forjando la Realidad (Script SQL Mejorado)

De Diagrama de Clases a Script SQL

-- Creamos el universo donde existirán nuestras tablas
CREATE DATABASE IF NOT EXISTS gestion_tareas;
USE gestion_tareas;

-- Tabla 'usuarios': Define quién puede actuar en el sistema.
CREATE TABLE usuarios (
    id_usuario INT PRIMARY KEY AUTO_INCREMENT,
    -- UNIQUE asegura que no haya dos usuarios con el mismo nombre. Crucial para logins.
    nombre_usuario VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100)
);

-- Tabla 'proyectos': Contenedores para nuestras tareas.
CREATE TABLE proyectos (
    id_proyecto INT PRIMARY KEY AUTO_INCREMENT,
    nombre_proyecto VARCHAR(100) NOT NULL,
    descripcion TEXT
);

-- Tabla 'tareas': El corazón del sistema. Conecta Proyectos y Usuarios.
CREATE TABLE tareas (
    id_tarea INT PRIMARY KEY AUTO_INCREMENT,
    titulo VARCHAR(255) NOT NULL,
    descripcion TEXT,
    -- CURRENT_DATE como valor por defecto para la fecha de creación.
    fecha_creacion DATE DEFAULT (CURRENT_DATE),
    fecha_vencimiento DATE,
    estado VARCHAR(20),
    
    id_proyecto INT,
    id_usuario_asignado INT,
    
    -- Creamos los "puentes" que aseguran la integridad de los datos.
    FOREIGN KEY (id_proyecto) REFERENCES proyectos(id_proyecto),
    FOREIGN KEY (id_usuario_asignado) REFERENCES usuarios(id_usuario)
);

El Secreto del DataGridView: El INNER JOIN

Para que el DataGridView en FormTareas sea útil, no puede mostrar solo IDs. Debe mostrar información legible. Aquí es donde la consulta SQL se vuelve arte:

SELECT 
    t.id_tarea, 
    t.titulo, 
    p.nombre_proyecto,  -- Traemos el nombre del proyecto
    u.nombre_usuario,   -- y el nombre del usuario
    t.fecha_vencimiento, 
    t.estado
FROM 
    tareas t
INNER JOIN proyectos p ON t.id_proyecto = p.id_proyecto
INNER JOIN usuarios u ON t.id_usuario_asignado = u.id_usuario;

Esta consulta une las tres tablas para crear una vista cohesiva y humana de los datos, lista para ser mostrada al usuario.
